\newglossaryentry{force}{
    name=force,
    description={a push or pull on an object with a specific magnitude and direction; can be represented by vectors; can be expressed as a multiple of a standard force}
}

\newglossaryentry{net force}{
    name=net force,
    description={the sum of all forces acting on an object or system}
}

\newglossaryentry{NIL}{
    name={Newton's First Law of Motion},
    description={a body at rest remains at rest or, if in motion, remains in motion at a constant speed in a straight line, unless acted on by a net external force; also known as the law of inertia}
}

\newglossaryentry{inertia}{
    name=inertia,
    description={the tendency of an object at rest to remain at rest, or for a moving object to remain in motion in a straight line and at a constant speed}
}

\newglossaryentry{NIIL}{
    name={Newton's Second Law of Motion},
    description={the net external force ($F_\mathrm{net}$) on an object is proportional to and in the same direction as the acceleration of the object ($a$) and also proportional to the objectâ€™s mass ($m$); defined mathematically as $F_\text{net}=ma$ or $\sum F = m a$.}
}

\newglossaryentry{friction}{
    name=friction,
    description={an external force that acts in the direction opposite to the direction of motion}
}

\newglossaryentry{free-body diagram}{
    name={free-body diagram},
    description={a diagram showing all external forces acting on a body}
}

\newglossaryentry{mass}{
    name=mass,
    description={the quantity of matter in a substance; measured in kilograms}
}

\newglossaryentry{equilibrium}{
    name=equilibrium,
    description={the state in which the net force on an object is zero; all forces are balanced, and the object does not accelerate}
}

\newglossaryentry{NIIIL}{
    name={Newton's Third Law of Motion},
    description={when one body exerts a force on a second body, the first body experiences a force that is equal in magnitude and opposite in direction to the force that it exerts}
}

\newglossaryentry{weight}{
    name=weight,
    description={($w$) the force of gravity acting on an object of mass $m$; defined mathematically as $W = mg$, where $g$ is the magnitude and direction of the acceleration due to gravity}
}

\newglossaryentry{normal force}{
    name=normal force,
    description={the force that a surface applies to an object; acts perpendicular and away from the surface with which the object is in contact}
}
