\makenoidxglossaries
\newglossaryentry{position}
{
    name = position,
    description = the location of an object on a fixed axis; the SI unit is the meter
}

\newglossaryentry{displacement}
{
    name=displacement,
    description=the net change in position between initial and final positions; the straight-line distance from the initial to the final position; asking ``What is the object's displacement?'' is like asking ``Where is the object relative to its original position?''
}

\newglossaryentry{distance}
{
    name=distance,
    description=the total length of the path traveled between two positions
}

\newglossaryentry{magnitude}{
    name=magnitude,
    description=size or amount; e.g. \textit{magnitude} of displacement
}

\newglossaryentry{scalar}{
    name=scalar,
    description={a physical quantity that has a magnitude but no direction; e.g., distance is a scalar}
}


\newglossaryentry{elapsed time}{
    name=elapsed time,
    description={the difference between the ending time and beginning time; also known as the change in time or the time interval}
}

\newglossaryentry{speed}{
    name=speed,
    description={rate at which an object changes its location}
}

\newglossaryentry{average speed}{
    name=average speed,
    description={distance traveled divided by time during which motion occurs}
}

\newglossaryentry{velocity}{
    name=velocity,
    description={the speed and direction of an object}
}

\newglossaryentry{average velocity}{
    name=average velocity,
    description={displacement divided by time over which displacement occurs}
}

\newglossaryentry{position vs. time graph}{
    name={position vs.~time graph},
    description={a graph where position is plotted on the $y$-axis and time is plotted on the $x$-axis}
}

\newglossaryentry{velocity vs. time graph}{
    name={velocity vs.~time graph},
    description={a graph where velocity is plotted on the $y$-axis and time is plotted on the $x$-axis}
}

\newglossaryentry{initial position}{
    name=initial position,
    description={the position of an object at the beginning of its motion}
}

\newglossaryentry{final position}{
    name=final position,
    description={the position of an object at the end of its motion}
}

\newglossaryentry{momentum}{
    name=momentum,
    description={the product of a system's mass and velocity}
}

\newglossaryentry{mass}{
    name=mass,
    description={the quantity of matter in a substance; measured in kilograms}
