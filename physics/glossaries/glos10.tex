\newglossaryentry{wave}
{
    name=wave,
    description=a disturbance that moves from a source and carries energy
}

\newglossaryentry{amplitude}{
    name=amplitude,
    description={the maximum displacement from the equilibrium position of an object oscillating around the equilibrium position}
}

\newglossaryentry{mechanical wave}{
    name=mechanical wave,
    description={wave that requires a medium through which it can travel}
}

\newglossaryentry{propagate}
{
    name=propagate,
    description=to move in a given direction
}

\newglossaryentry{transverse wave}
{
    name=transverse wave,
    description=a wave in which the disturbance is \textit{perpendicular} to the direction of propagation
}

\newglossaryentry{medium}{
    name=medium,
    description={solid, liquid, or gas material through which a wave propagates}
}

\newglossaryentry{crest}
{
    name=crest,
    description=the highest position in the cycle of a wave
}

\newglossaryentry{trough}
{
    name=trough,
    description=the lowest position in the cycle of a wave
}

\newglossaryentry{adjacent}
{
    name=adjacent,
    description={next to; for example, a person sitting next to you is said to be ``adjacent'' to you}
}

\newglossaryentry{wavelength}
{
    name=wavelength,
    description=the distance between adjacent identical parts of a wave
}

\newglossaryentry{wave cycle}{
    name = wave cycle,
    description= the potion of a wave encompassed by 1 wavelength
}

\newglossaryentry{oscillate}{
    name = oscillate,
    description = moving back and forth regularly between two points
}

\newglossaryentry{period}{
    name = period,
    description = the time it takes to complete one wave cycle or oscillation
}

\newglossaryentry{frequency}{
    name = frequency,
    description = the number of wave cycles or oscillations completed in a time interval of 1 second
}

\newglossaryentry{wave velocity}{
    name = wave velocity,
    description = the speed at which a wave moves; also known as propagation velocity or propagation speed
}

\newglossaryentry{superposition}{
    name=superposition,
    description={summation phenomenon that occurs when two waves arrive at the same point}
}

\newglossaryentry{longitudinal wave}{
    name = longitudinal wave,
    description = a wave in which the disturbance is \textit{parallel} to the direction of propagation
}


\newglossaryentry{rarefaction}
{
    name = rarefaction,
    description = {a low-pressure region in a sound wave}
}

\newglossaryentry{sound}{
    name = sound,
    description={a disturbance of matter that is transmitted from its source outward by longitudinal waves}
}

\newglossaryentry{compression}
{
    name=compression,
    description={a high-pressure region in a sound wave}
}

\newglossaryentry{rigid}
{
    name = rigid,
    description={unable to bend or be forced out of shape}
}

\newglossaryentry{rigidity}
{
    name = rigidity,
    description={a property that allows a substance to remain stiff and unbent},
}

\newglossaryentry{compress}
{
    name = compressibility,
    description={to flatten or squeeze a gas by pressure}
}

\newglossaryentry{compressibility}
{
    name = compressibility,
    description={a property that allows a substance to be squeezed into a smaller volume}
}

\newglossaryentry{density}
{
    name=density,
    description={the mass of a substance per unit volume; the amount of matter contained per unit volume}
}


\newglossaryentry{pitch}
{
    name = pitch,
    description = {the brain's perception of the frequency of a sound}
}

\newglossaryentry{pressure}
{
    name=pressure,
    description={force per unit area perpendicular to the force, over which the force acts}
}

\newglossaryentry{atmospheric pressure}
{
    name={atmospheric pressure},
    description={the pressure of the air under normal, atmospheric conditions; measured as \SI{101325}{Pa} or \SI{1}{atm}}
}

\newglossaryentry{gauge pressure}
{
    name={gauge pressure},
    description={the pressure relative to atmospheric pressure; it is positive for pressures above atmospheric pressure, and negative for pressures below it}
}

\newglossaryentry{constructive interference}{
    name=constructive interference,
    description={when two waves arrive at the same point exactly in phase; that is, the crests of the two waves are precisely aligned, as are the troughs}
}

\newglossaryentry{electric field}{
    name=electric field,
    description={a field that tells us the force per unit charge at all locations in space around a charge distribution}
}

\newglossaryentry{emr}{
    name={electromagnetic radiation (EMR)},
    description={radiant energy that consists of oscillating electric and magnetic fields}
}
