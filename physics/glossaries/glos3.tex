\newglossaryentry{acceleration}{
    name=acceleration,
    description={rate at which velocity changes}
}

\newglossaryentry{average acceleration}{
    name=average acceleration,
    description={change in velocity divided by the time interval over which it changed}
}

\newglossaryentry{instantaneous acceleration}{
    name=instantaneous acceleration,
    description={rate of change of velocity at a specific instant in time}
}

\newglossaryentry{negative acceleration}{
    name=negative acceleration,
    description={a component of acceleration in the negative direction of the component axis}
}
