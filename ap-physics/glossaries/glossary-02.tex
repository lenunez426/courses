\makenoidxglossaries

\newglossaryentry{position}{
    name=position,
    description={the location of an object at a particular time}
}

\newglossaryentry{displacement}{
    name=displacement,
    description={the change in position of an object}
}

\newglossaryentry{distance}{
    name=distance,
    description={the magnitude of displacement between two positions}
}

\newglossaryentry{distance traveled}{
    name=distance traveled,
    description={the total length of the path traveled between two positions}
}

\newglossaryentry{vector}{
    name=vector,
    description={a quantity that is described by both magnitude and direction}
}

\newglossaryentry{scalar}{
    name=scalar,
    description={a quantity that is described by magnitude, but not direction}
}

\newglossaryentry{time}{
    name=time,
    description={change, or the interval over which change occurs}
} 

\newglossaryentry{elapsed time}{
    name=elapsed time,
    description={the difference between the ending time and beginning time}
}

\newglossaryentry{average velocity}{
    name=average velocity,
    description={displacement divided by time over which displacement occurs}
}

\newglossaryentry{instantaneous velocity}{
    name=instantaneous velocity,
    description={velocity at a specific instant, or the average velocity over an infinitesimal time interval}
}

\newglossaryentry{instantaneous speed}{
    name=instantaneous speed,
    description={magnitude of the instantaneous velocity}
}

\newglossaryentry{average speed}{
    name=average speed,
    description={distance traveled divided by time during which motion occurs}
}

\newglossaryentry{model}{
    name=model,
    description={simplified description that contains only those elements necessary to describe the physics of a physical situation}
}
