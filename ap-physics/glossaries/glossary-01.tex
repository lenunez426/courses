\makenoidxglossaries

\newglossaryentry{physical quantity}{
    name=physical quantity,
    description={a characteristic or property of an object that can be measured or calculated from other measurements}
}

\newglossaryentry{units}{
    name=units,
    description={a standard used for expressing and comparing measurements}
}

\newglossaryentry{SI units}{
    name=SI units,
    description={the international system of units that scientists in most countries have agreed to use; includes units such as meters, liters, and grams}
}

\newglossaryentry{English units}{
    name=English units,
    description={system of measurement used in the United States; includes units of measurement such as feet, gallons, and pounds}
}

\newglossaryentry{fundamental units}{
    name=fundamental units,
    description={units that can only be expressed relative to the procedure used to measure them}
}

\newglossaryentry{derived units}{
    name=derived units,
    description={units that can be calculated using algebraic combinations of the fundamental units}
}

\newglossaryentry{second}{
    name=second,
    description={the SI unit for time, abbreviated (s)}
}

\newglossaryentry{meter}{
    name=meter,
    description={the SI unit for length, abbreviated (m)}
}

\newglossaryentry{kilogram}{
    name=kilogram,
    description={the SI unit for mass, abbreviated (kg)}
}

\newglossaryentry{metric system}{
    name=metric system,
    description={a system in which values can be calculated in factors of 10}
}

\newglossaryentry{order of magnitude}{
    name=order of magnitude,
    description={refers to the size of a quantity as it relates to a power of 10}
}

\newglossaryentry{conversion factor}{
    name=conversion factor,
    description={a ratio expressing how many of one unit are equal to another unit}
}
