\makenoidxglossaries

\newglossaryentry{dynamics}{
    name={dynamics},
    description={the study of how forces affect the motion of objects and systems}
}

\newglossaryentry{force}{
    name={force},
    description={a push or pull on an object with a specific magnitude and direction; can be represented by vectors; can be expressed as a multiple of a standard force}
}

\newglossaryentry{free-body diagram}{
    name={free-body diagram},
    description={a sketch showing all of the external forces acting on an object or system; the system is represented by a dot, and the forces are represented by vectors extending outward from the dot}
}

\newglossaryentry{external force}{
    name={external force},
    description={a force acting on an object or system that originates outside of the object or system}
}

\newglossaryentry{Newton's first law of motion}{
    name={Newton's first law of motion},
    description={a body at rest remains at rest, or, if in motion, remains in motion at a constant velocity unless acted on by a net external force; also known as the law of inertia}
}

\newglossaryentry{law of inertia}{
    name={law of inertia},
    description={see Newton's first law of motion}
}

\newglossaryentry{mass}{
    name={mass},
    description={the quantity of matter in a substance; measured in kilograms}
}

\newglossaryentry{inertia}{
    name={inertia},
    description={the tendency of an object to remain at rest or remain in motion}
}

\newglossaryentry{Newton's second law of motion}{
    name={Newton's second law of motion},
    description={the net external force $\textbf{F}_{\text{net}}$ on an object with mass $m$ is proportional to and in the same direction as the acceleration of the object, \textbf{a}, and inversely proportional to the mass; defined mathematically as $\textbf{a} = \frac{\textbf{F}_{\text{net}}}{m}$.}
}

\newglossaryentry{acceleration}{
    name={acceleration},
    description={the rate at which an objectâ€™s velocity changes over a period of time}
}

\newglossaryentry{system}{
    name={system},
    description={defined by the boundaries of an object or collection of objects being observed; all forces originating from outside of the system are considered external forces}
}

\newglossaryentry{friction}{
    name={friction},
    description={a force past each other of objects that are touching; examples include rough surfaces and air resistance}
}

\newglossaryentry{net external force}{
    name={net external force},
    description={the vector sum of all external forces acting on an object or system; causes a mass to accelerate}
}

\newglossaryentry{newton}{
    name={newton},
    description={the SI unit of force}
}

\newglossaryentry{weight}{
    name={weight},
    description={the force \textbf{w} due to gravity acting on an object of mass $m$; defined mathematically as: $\textbf{w} = m \textbf{g}$, where \textbf{g} is the magnitude and direction of the acceleration due to gravity}
}
