\tikzmath{
        \a = 1;
        \e = 0.5;
        \b = \a*sqrt(1-\e^2);
        \c = sqrt(\a^2 - \b^2);
        \fone = 2*pi - 5.9341;
        \ftwo = 0.3348;
        \fthree = 3.10309;
        \ffour = 3.1801;
    }
\begin{axis}[width=12cm,
    xmin=-1,xmax=1,
    ymin=-1,ymax=1,
    clip=false,
    axis lines = center,
    % axis line style={draw=none},
    % ticks=none,
    axis equal image,
    grid=both
    ]
    \draw[thick] (0,0) ellipse ({\a} and {\b});
    \fill[yellow!80] (0:\c) -- (0:\a) arc (0:-\fone r:{\a} and {\b}) -- cycle;
    \fill[yellow!80] (0:\c) -- (0:\a) arc (0:\ftwo r:{\a} and {\b}) -- cycle;
    \fill[yellow!80] (0:\c) -- (pi r:\a) arc (pi r:\fthree r:{\a} and {\b}) -- cycle;
    \fill[yellow!80] (0:\c) -- (pi r:\a) arc (pi r:\ffour r:{\a} and {\b}) -- cycle;
    \draw[black,fill=yellow=80!] (0:\c) circle (2mm) node[above=2mm] {Sun};

\end{axis}
