\makenoidxglossaries


\newglossaryentry{orbit}{
    name={orbit},
    description={the path of an object that is in revolution about another object or point}
}

\newglossaryentry{ellipse}{
    name={ellipse},
    description={a closed curve for which the sum of the distances from any point on the ellipse to two points inside (called the foci) is always the same}
}

\newglossaryentry{focus}{
    name={focus},
    description={(plural: foci) one of two fixed points inside an ellipse from which the sum of the distances to any point on the ellipse is constant}
}

\newglossaryentry{major axis}{
    name={major axis},
    description={the maximum diameter of an ellipse}
}

\newglossaryentry{semimajor axis}{
    name={semimajor axis},
    description={half of the major axis of a conic section, such as an ellipse}
}

\newglossaryentry{eccentricity}{
    name={eccentricity},
    description={in an ellipse, the ratio of the distance between the foci to the major axis}
}

\newglossaryentry{orbital speed}{
    name={orbital speed},
    description={the speed at which an object (usually a planet) orbits around the mass of another object; in the case of a planet, the speed at which each planet moves along its ellipse}
}

\newglossaryentry{orbital period}{
    name={orbital period},
    description={the time it takes an object to travel once around the Sun}
}

\newglossaryentry{astronomical unit}{
    name={astronomical unit},
    description={the unit of length defined as the average distance between Earth and the Sun; this distance is about \SI{1.5e8}{} kilometers}
}

\newglossaryentry{Kepler's first law}{
    name={Kepler's first law},
    description={each planet moves around the Sun in an orbit that is an ellipse, with the Sun at one focus of the ellipse}
}

\newglossaryentry{Kepler's second law}{
    name={Kepler's second law},
    description={the straight line joining a planet and the Sun sweeps out equal areas in space in equal intervals of time}
}

\newglossaryentry{Kepler's third law}{
    name={Kepler's third law},
    description={the square of a planet's orbital period is directly proportional to the cube of the semimajor axis of its orbit}
}

\newglossaryentry{Newton's first law}{
    name={Newton's first law},
    description={every object will continue to be in a state of rest or move at a constant speed in a straight line unless it is compelled to change by an outside force}
}

\newglossaryentry{Newton's second law}{
    name={Newton's second law},
    description={the change of motion of a body is proportional to and in the direction of the force acting on it}
}

\newglossaryentry{Newton's third law}{
    name={Newton's third law},
    description={for every action there is an equal and opposite reaction (or: the mutual actions of two bodies upon each other are always equal and act in opposite directions)}
}

\newglossaryentry{momentum}{
    name={momentum},
    description={the measure of the amount of motion of a body; the momentum of a body is the product of its mass and velocity; in the absence of an unbalanced force, momentum is conserved}
}

\newglossaryentry{velocity}{
    name={velocity},
    description={the speed and direction a body is moving---for example, 44 kilometers per second toward the north galactic pole}
}

\newglossaryentry{mass}{
    name={mass},
    description={a measure of the amount of material within an object}
}

\newglossaryentry{density}{
    name={density},
    description={the ratio of the mass of an object to its volume}
}

\newglossaryentry{angular momentum}{
    name={angular momentum},
    description={the measure of the motion of a rotating object in terms of its speed and how widely the object's mass is distributed around its axis}
}
