\newglossaryentry{aurora}{
    name=aurora,
    description={light radiated by atoms and ions in the ionosphere excited by charged particles from the Sun, mostly seen in the magnetic polar regions}
}

\newglossaryentry{corona}{
    name=corona,
    description={(of the Sun) the outer (hot) atmosphere of the Sun}
}

\newglossaryentry{sunspot}{
    name=sunspot,

}

\newglossaryentry{plasma}{
    name=plasma,
    description={a hot ionized gas}
}

\newglossaryentry{photosphere}{
    name=photosphere,
    description={the region of the solar (or stellar) atmosphere from which continuous radiation escapes into space}
}

    name=chromosphere,
    description={the part of the solar atmosphere that lies immediately above the photospheric layers}
}
