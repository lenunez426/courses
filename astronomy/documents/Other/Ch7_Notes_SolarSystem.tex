\makeglossaries

\newglossaryentry{terrestrial planet}{
    name=terrestrial planet,
    description={any of the planets Mercury, Venus, Earth, or Mars; sometimes the Moon is included in the list}
}

\newglossaryentry{giant planet}{
    name=giant planet,
    description={any of the planets Jupiter, Saturn, Uranus, and Neptune in our solar system, or planets of roughly that mass and composition in other planetary systems}
}

\newglossaryentry{comet}{
    name=comet,
    description={a small body of icy and dusty matter that revolves about the Sun; when a comet comes near the Sun, some of its material vaporizes, forming a large head of tenuous gas and often a tail}
}

\newglossaryentry{asteroid}{
    name=asteroid,
    description={a stony or metallic object orbiting the Sun that is smaller than a planet but that shows no evidence of an atmosphere or of other types of activity associated with comets}
}
