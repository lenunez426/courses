\documentclass{article}
\usepackage{preamble}

\makeglossaries

\newglossaryentry{Big Bang}{
    name=Big Bang,
    description={the theory of cosmology in which the expansion of the universe began with a primeval explosion (of space, time, matter, and energy)}
}

\newglossaryentry{CMB}{
    name=cosmic microwave background (CMB),
    description={microwave radiation coming from all directions that is the redshifted afterglow of the Big Bang}
}

\newglossaryentry{cosmology}{
    name=cosmology,
    description={the study of the organization and evolution of the universe}
}

\newglossaryentry{dark energy}{
    name=dark energy,
    description={the energy that is causing the expansion of the universe to accelerate; its existence is inferred from observations of distant supernovae}
}

\newglossaryentry{dark matter}{
    name=dark matter,
    description={nonluminous material, whose nature we don't yet understand, but whose presence can be inferred because of its gravitational influence on luminous matter}
}

\newglossaryentry{GUTs}{
    name=grand unified theories (GUTs),
    description={physical theories that attempt to describe the four forces of nature as different manifestations of a single force}
}

\newglossaryentry{lithium}{
    name=lithium,
    description={the third element in the periodic table; lithium nuclei with three protons and four neutrons were manufactured during the first few minutes of the expansion of the universe}
}

\newglossaryentry{multiverse}{
    name=multiverse,
    description={the speculative idea that our universe is just one of many universes, each with its own set of physical laws}
}

\newglossaryentry{cosmological constant}{
    name=cosmological constant ($\Lambda$),
    description={the term in the equations of general relativity that represents a repulsive force in the universe}
}

\newglossaryentry{deuterium}{
    name=deuterium,
    description={a form of hydrogen in which the nucleus of each atom consists of one proton and one neutron}
}
