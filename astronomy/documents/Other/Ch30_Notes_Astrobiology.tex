\documentclass{article}
\usepackage{preamble}

\makeglossaries

\newglossaryentry{amino acids}{
    name=amino acids,
    description={organic compounds that are the molecular building blocks of proteins}
}

\newglossaryentry{DNA}{
    name=DNA (deoxyribonucleic acid),
    description={a molecule that stores information about how to replicate a cell and its chemical and structural components}
}

\newglossaryentry{Drake equation}{
    name=Drake equation,
    description={a formula for estimating the number of intelligent, technological civilizations in our Galaxy, first suggested by Frank Drake}
}

\newglossaryentry{habitable zone}{
    name=habitable zone,
    description={the region around a star in which liquid water could exist on the surface of terrestrial-sized planets, hence the most probable place to look for life in a star's planetary system}
}

\newglossaryentry{organic compound}{
    name=organic compound,
    description={a compound containing carbon, especially a complex carbon compound; not necessarily produced by life}
}

\newglossaryentry{SETI}{
    name=SETI,
    description={the search for extraterrestrial intelligence; usually applied to searches for radio signals from other civilizations}
}

\newglossaryentry{organic molecule}{
    name=organic molecule,
    description={a combination of carbon and other atoms—primarily hydrogen, oxygen, nitrogen, phosphorus, and sulfur—some of which serve as the basis for our biochemistry}
}

\newglossaryentry{astrobiology}{
    name=astrobiology,
    description={the multidisciplinary study of life in the universe: its origin, evolution, distribution, and fate; similar terms are exobiology and bioastronomy}
}

\newglossaryentry{protein}{
    name=protein,
    description={a key biological molecule that provides the structure and function of the body’s tissues and organs, and essentially carries out the chemical work of the cell}
}

\newglossaryentry{gene}{
    name=gene,
    description={the basic functional unit that carries the genetic (hereditary) material contained in a cell}
}

\newglossaryentry{photosynthesis}{
    name=photosynthesis,
    description={a complex sequence of chemical reactions through which some living things can use sunlight to manufacture products that store energy (such as carbohydrates), releasing oxygen as one by-product}
}

\end{document}
