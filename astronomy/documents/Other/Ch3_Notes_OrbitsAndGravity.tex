\documentclass{article}
\usepackage{preamble}

\makeglossaries 
\newglossaryentry{AU}
{
    name=astronomical unit,
    description={the unit of length defined as the average distance between Earth and the Sun; this distance is about \SI{1.5e8}{} kilometers; astronomical unit is often abbreviated AU}
}

\newglossaryentry{ellipse}{
    name=ellipse,
    description={a closed curve for which the sum of the distances from any point on the ellipse to two points inside (called the foci) is always the same}
}

\newglossaryentry{eccentricity}{
    name=eccentricity,
    description={in an ellipse, the ratio of the distance between the foci to the major axis}
}

\newglossaryentry{escape speed}{
    name=escape speed,
    description={the speed a body must achieve to break away from the gravity of another body}
}

\newglossaryentry{gravity}{
    name=gravity,
    description={the mutual attraction of material bodies or particles}
}

\newglossaryentry{mass}{
    name=mass,
    description={a measure of the amount of material within an object}
}

\newglossaryentry{orbit}{
    name=orbit,
    description={the path of an object that is in revolution about another object or point}
}

\newglossaryentry{orbital period}{
    name=orbital period,
    description={the time it takes an object to travel once around the Sun; abbreviated $P$}
}

\newglossaryentry{orbital speed}{
    name=orbital speed,
    description={the speed at which an object (usually a planet) orbits around the mass of another object; in the case of a planet, the speed at which each planet moves along its ellipse}
}

\newglossaryentry{velocity}{
    name=velocity,
    description={the speed and direction a body is moving; for example, 44 kilometers per second toward the north galactic pole}
}

\clearpage

\printglossaries
\end{document}
