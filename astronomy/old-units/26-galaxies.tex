\makenoidxglossaries

\newglossaryentry{central bulge}{
    name=central bulge,
    description={(or nuclear bulge) the central (round) part of the Milky Way or a similar galaxy}
}

\newglossaryentry{dark matter}{
    name=dark matter,
    description={nonluminous mass, whose presence can be inferred only because of its gravitational influence on luminous matter; the composition of the dark matter is not known}
}

\newglossaryentry{Milky Way Galaxy}{
    name=Milky Way Galaxy,
    description={the band of light encircling the sky, which is due to the many stars and diffuse nebulae lying near the plane of the Milky Way Galaxy}
}

\newglossaryentry{spiral arm}{
    name=spiral arm,
    description={a spiral-shaped region, characterized by relatively dense interstellar material and young stars, that is observed in the disks of spiral galaxies}
}

\newglossaryentry{supermassive black hole}{
    name=supermassive black hole,
    description={the object in the center of most large galaxies that is so massive and compact that light cannot escape from it; the Milky Way’s supermassive black hole contains 4.6 millions of Suns’ worth of mass}
}

\newglossaryentry{elliptical galaxy}{
    name=elliptical galaxy,
    description={a galaxy whose shape is an ellipse and that contains no conspicuous interstellar material}
}

\newglossaryentry{Hubble's law}{
    name=Hubble's law,
    description={a rule that the radial velocities of remote galaxies are proportional to their distances from us}
}

\newglossaryentry{irregular galaxy}{
    name=irregular galaxy,
    description={a galaxy without any clear symmetry or pattern; neither a spiral nor an elliptical galaxy}
}

\newglossaryentry{redshift}{
    name=redshift,
    description={when lines in the spectra are displaced toward longer wavelengths (toward the red end of the visible spectrum)}
}

\newglossaryentry{spiral galaxy}{
    name=spiral galaxy,
    description={a flattened, rotating galaxy with pinwheel-like arms of interstellar material and young stars, winding out from its central bulge}
}

\newglossaryentry{type Ia supernova}{
    name=type Ia supernova,
    description={a supernova formed by the explosion of a white dwarf in a binary system and reach a luminosity of about \SI{4.5e9}{L_{Sun}}; can be used to determine distances to galaxies on a large scale}
}

\newglossaryentry{Hubble constant}{
    name=Hubble constant,
    description={a constant of proportionality in the law relating the velocities of remote galaxies to their distances}
}
